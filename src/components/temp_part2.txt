            <h2 className="text-lg font-bold text-gray-900">Informações do Story</h2>
            <button onClick={onClose} className="text-gray-600 hover:text-gray-800 transition-colors">
              <X className="w-6 h-6" />
            </button>
          </div>

          {/* Conteúdo */}
          <div className="p-6">
            <div className="text-center mb-8">
              {/* Título */}
              <h3 className="text-2xl font-bold text-gray-900 mb-2">
                Informações do Story
              </h3>
              <p className="text-gray-600 text-sm">
                ✨ Quase lá! Preencha com seus dados:
              </p>
            </div>

            {/* Formulário */}
            <div className="space-y-6">
              {/* Nome */}
              <div>
                <label className="block text-gray-900 font-medium mb-2 text-sm">
                  Nome
                </label>
                <input
                  type="text"
                  value={companionName}
                  onChange={(e) => setCompanionName(e.target.value)}
                  placeholder="Digite seu nome"
                  className="w-full px-4 py-3 rounded-xl bg-white border border-gray-200 text-gray-900 placeholder-gray-400 focus:border-pink-400 focus:outline-none focus:ring-2 focus:ring-pink-100 text-sm font-sans"
                  required
                />
                <p className="text-gray-500 text-xs mt-1">
                  Como será exibido no story
                </p>
              </div>

              {/* WhatsApp */}
              <div>
                <label className="block text-gray-900 font-medium mb-2 text-sm">
                  WhatsApp
                </label>
                <div className="flex">
                  <span className="inline-flex items-center px-3 rounded-l-xl border border-r-0 border-gray-200 bg-gray-50 text-gray-600 text-sm">
                    +55
                  </span>
                  <input
                    type="tel"
                    value={companionWhatsapp}
                    onChange={(e) => setCompanionWhatsapp(e.target.value)}
                    placeholder="Digite seu número"
                    className="flex-1 rounded-r-xl border border-gray-200 px-4 py-3 text-gray-900 placeholder-gray-400 focus:border-pink-400 focus:outline-none focus:ring-2 focus:ring-pink-100 text-sm font-sans"
                  />
                </div>
                <p className="text-gray-500 text-xs mt-1">
                  Usado para contato e como botão no seu story
                </p>
              </div>

              {/* Cidade */}
              <div>
                <label className="block text-gray-900 font-medium mb-2 text-sm">
                  Cidade
                </label>
                <CityAutocomplete
                  value={companionCity}
                  onChange={setCompanionCity}
                  placeholder="Digite sua cidade"
                  className="w-full px-4 py-3 rounded-xl bg-white border border-gray-200 text-gray-900 placeholder-gray-400 focus:border-pink-400 focus:outline-none focus:ring-2 focus:ring-pink-100 text-sm font-sans !text-gray-900 !placeholder-gray-400"
                />
                <p className="text-gray-500 text-xs mt-1">
                  Onde você atua
                </p>
              </div>

            </div>

            {/* Botões */}
            <div className="mt-8 space-y-4">
              {isUploading ? (
                <div className="w-full py-4 bg-gray-300 text-gray-700 rounded-full text-center">
                  <div className="flex items-center justify-center space-x-2">
                    <div className="animate-spin rounded-full h-5 w-5 border-b-2 border-gray-700"></div>
                    <span>Enviando story...</span>
                  </div>
                </div>
              ) : (
                <button
                  onClick={handleUpload}
                  disabled={!companionName || !companionWhatsapp || !companionCity}
                  className="w-full py-3 bg-pink-100 text-pink-700 rounded-full font-bold text-sm transition-all disabled:opacity-50 disabled:cursor-not-allowed"
                  style={{ backgroundColor: '#fcd3df', color: '#d63384' }}
                >
                  🚀 Enviar Story
                </button>
              )}
              
              <button
                onClick={() => setCurrentStep('preview')}
                className="w-full text-center text-gray-600 hover:text-gray-800 transition-colors text-sm font-light"
              >
                ← Voltar
              </button>
            </div>
          </div>
        </div>
      </div>
    );
  }

  // Renderizar etapa de criação
  return (
    <div className="fixed inset-0 bg-black bg-opacity-90 flex items-center justify-center z-50">
      <div className="relative w-full h-full max-w-md mx-auto bg-black flex flex-col">
        {/* Header */}
        <div className="absolute top-0 left-0 right-0 z-20 flex justify-between items-center p-4">
          <button
            onClick={() => setCurrentStep('plans')}
            className="text-white hover:text-gray-300 transition-colors"
          >
            <ArrowLeft className="w-6 h-6" />
          </button>
          
          <button
            onClick={onClose}
            className="text-white hover:text-gray-300 transition-colors"
          >
            <X className="w-6 h-6" />
          </button>
          
          {/* Timer para gravação */}
          {(isRecording || isVideoRecording) && (
            <div className="bg-red-500 text-white px-3 py-1 rounded-full text-sm font-bold">
              {Math.floor(recordingTime / 60)}:{(recordingTime % 60).toString().padStart(2, '0')}
            </div>
          )}
        </div>

        {/* Área principal */}
        <div className="flex-1 relative overflow-hidden">
          {/* Câmera ativa */}
          {storyType === 'camera' || storyType === 'video' ? (
            !preview ? (
              <div className="w-full h-full relative">
                <video
                  ref={cameraVideoRef}
                  autoPlay
                  playsInline
                  muted
                  className={`w-full h-full object-cover ${facingMode === 'user' ? 'transform -scale-x-100' : ''}`}
                  style={{ transform: facingMode === 'user' ? 'scaleX(-1)' : 'none' }}
                />
                
                {/* Controles da câmera */}
                <div className="absolute bottom-20 left-0 right-0 flex flex-col items-center">
                  {/* Contador de tempo */}
                  {isVideoRecording && (
                    <div className="mb-4 px-4 py-2 bg-black/50 rounded-full text-white text-lg">
                      {recordingTime}s
                    </div>
                  )}
                  
                  {/* Botões */}
                  <div className="flex justify-center items-center space-x-8">
                  {/* Botão alternar câmera */}
                  <button
                    onClick={switchCamera}
                    className="p-3 bg-black/50 rounded-full text-white hover:bg-black/70 transition-colors"
                      title="Alternar câmera"
                  >
                    <FlipHorizontal className="w-6 h-6" />
                  </button>

                  {/* Botão capturar/gravar */}
                  <button
                    onClick={storyType === 'camera' ? capturePhoto : (isVideoRecording ? stopVideoRecording : startVideoRecording)}
                      className={`relative flex items-center justify-center transition-all ${
                      storyType === 'camera' 
                          ? 'bg-white hover:bg-gray-200 w-16 h-16 rounded-full'
                        : isVideoRecording 
                            ? 'bg-red-500 hover:bg-red-600 w-12 h-12 rounded-lg'
                            : 'bg-red-500 hover:bg-red-600 w-16 h-16 rounded-full'
                    }`}
                      title={storyType === 'camera' ? 'Tirar foto' : (isVideoRecording ? 'Parar gravação' : 'Iniciar gravação')}
                  >
                    {storyType === 'camera' ? (
                      <Camera className="w-8 h-8 text-black" />
                    ) : isVideoRecording ? (
                        <Square className="w-6 h-6 text-white" />
                    ) : (
                      <Video className="w-8 h-8 text-white" />
                    )}
                      
                      {/* Indicador de gravação */}
                      {isVideoRecording && (
                        <div className="absolute -top-1 -right-1 w-3 h-3 rounded-full bg-red-500 animate-pulse" />
                      )}
                  </button>

                  {/* Placeholder para simetria */}
                  <div className="p-3 w-12 h-12"></div>
                </div>
                </div>

                {/* Dicas de uso */}
                {storyType === 'video' && !isVideoRecording && !preview && (
                  <div className="absolute top-4 left-1/2 transform -translate-x-1/2 bg-black/50 px-4 py-2 rounded-full text-white text-sm">
                    Toque no botão para começar a gravar
                  </div>
                )}
                
                {isVideoRecording && (
                  <div className="absolute top-4 left-1/2 transform -translate-x-1/2 bg-black/50 px-4 py-2 rounded-full text-white text-sm">
                    Toque novamente para parar
                  </div>
                )}
              </div>
            ) : (
              <div className="w-full h-full relative">
                {file?.type.startsWith('video/') ? (
                  <video
                    ref={videoRef}
                    src={preview}
                    controls
                    className="w-full h-full object-cover"
                  />
                ) : (
                  <img
                    src={preview}
                    alt="Preview"
                    className="w-full h-full object-cover"
                  />
                )}
              </div>
            )
          ) : storyType === 'audio' ? (
            <div className="w-full h-full bg-gradient-to-br from-purple-500 to-pink-500 flex flex-col items-center justify-center p-8">
              <div className="text-center text-white mb-8">
                <Mic className="w-16 h-16 mx-auto mb-4" />
                <h3 className="text-xl font-bold mb-2">Story de Áudio</h3>
                <p className="text-white/80">Grave até 30 segundos</p>
              </div>

              {!preview ? (
                <div className="text-center">
                  <button
                    onClick={isRecording ? stopAudioRecording : startAudioRecording}
                    className={`p-6 rounded-full transition-all ${
                      isRecording 
                        ? 'bg-red-500 hover:bg-red-600 animate-pulse' 
                        : 'bg-white/20 hover:bg-white/30'
                    }`}
                  >
                    {isRecording ? (
                      <Square className="w-8 h-8 text-white" />
                    ) : (
                      <Mic className="w-8 h-8 text-white" />
                    )}
                  </button>
                  
                  {isRecording && (
                    <div className="mt-4 text-white text-lg font-bold">
                      {Math.floor(recordingTime / 60)}:{(recordingTime % 60).toString().padStart(2, '0')}
                    </div>
                  )}
                </div>
              ) : (
                <div className="text-center">
                  <div className="mb-4">
                    <div className="w-20 h-20 bg-white/20 rounded-full flex items-center justify-center mx-auto mb-4">
                      <Mic className="w-10 h-10 text-white" />
                    </div>
                    <p className="text-white">Áudio gravado: {recordingTime}s</p>
                  </div>
                  
                  <div className="flex space-x-4">
                    <button
                      onClick={isPlaying ? pauseAudio : playAudio}
                      className="p-3 bg-white/20 rounded-full hover:bg-white/30 transition-colors"
                    >
                      {isPlaying ? (
                        <Pause className="w-6 h-6 text-white" />
                      ) : (
                        <Play className="w-6 h-6 text-white" />
                      )}
                    </button>
                    
                    <button
                      onClick={() => {
                        setPreview(null);
                        setFile(null);
                      }}
                      className="p-3 bg-white/20 rounded-full hover:bg-white/30 transition-colors"
                    >
                      <RotateCcw className="w-6 h-6 text-white" />
                    </button>
                  </div>

                  <audio
                    ref={audioRef}
                    src={preview || ''}
                    onEnded={() => setIsPlaying(false)}
                    className="hidden"
                  />
                </div>
              )}
            </div>
          ) : storyType === 'text' ? (
            <div className="w-full h-full relative" style={{ backgroundColor }}>
              <div className="absolute inset-0 flex items-center justify-center p-8">
                <textarea
                  value={text}
                  onChange={(e) => setText(e.target.value)}
                  placeholder="Digite seu texto aqui..."
                  className="w-full h-32 bg-transparent text-center resize-none border-none outline-none text-2xl font-bold"
                  style={{ color: textColor }}
                  maxLength={200}
                />
              </div>

              {/* Paleta de cores */}
              <div className="absolute bottom-32 left-0 right-0 p-4">
                <div className="mb-4">
                  <p className="text-white/90 text-sm mb-2 text-center">Cor do fundo:</p>
                  <div className="flex justify-center space-x-2">
                    {backgroundColors.map((color) => (
                      <button
                        key={color}
                        onClick={() => setBackgroundColor(color)}
                        className={`w-8 h-8 rounded-full border-2 ${
                          backgroundColor === color ? 'border-white' : 'border-transparent'
                        }`}
                        style={{ backgroundColor: color }}
                      />
                    ))}
                  </div>
                </div>
                
                <div>
                  <p className="text-white/90 text-sm mb-2 text-center">Cor do texto:</p>
                  <div className="flex justify-center space-x-2">
                    {textColors.map((color) => (
                      <button
                        key={color}
                        onClick={() => setTextColor(color)}
                        className={`w-6 h-6 rounded-full border-2 ${
                          textColor === color ? 'border-white' : 'border-gray-400'
                        }`}
                        style={{ backgroundColor: color }}
                      />
                    ))}
                  </div>
                </div>
              </div>
            </div>
          ) : storyType === 'upload' ? (
            <div className="w-full h-full bg-gray-900 flex flex-col items-center justify-center p-8">
              {!preview ? (
                <div className="text-center text-white">
                  <Upload className="w-16 h-16 mx-auto mb-4" />
                  <h3 className="text-xl font-bold mb-4">Upload de Arquivo</h3>
                  <button
                    onClick={() => fileInputRef.current?.click()}
                    className="px-6 py-3 bg-velvet-pink-600 text-white rounded-lg hover:bg-velvet-pink-700 transition-colors"
                  >
                    Escolher Arquivo
                  </button>
                  <p className="text-gray-400 text-sm mt-2">
                    Imagens até 15MB ou vídeos até 50MB
                  </p>
                </div>
              ) : (
                <div className="w-full h-full relative">
                  {file?.type.startsWith('video/') ? (
                    <video
                      ref={videoRef}
                      src={preview}
                      controls
                      className="w-full h-full object-cover"
                    />
                  ) : (
                    <img
                      src={preview}
                      alt="Preview"
                      className="w-full h-full object-cover"
                    />
                  )}
                </div>
              )}
            </div>
          ) : null}
        </div>

        {/* Barra lateral de tipos de story */}
        <div className="absolute right-4 top-1/2 transform -translate-y-1/2 z-20">
          <div className="flex flex-col space-y-4">
            {storyTypes.map((type) => (
              <button
                key={type.type}
                onClick={() => {
                  setStoryType(type.type);
                  setPreview(null);
                  setFile(null);
                  
                  if (type.type === 'camera' || type.type === 'video') {
                    startCamera();
                  } else {
                    stopCamera();
                  }
                }}
                className={`p-3 rounded-full transition-all ${
                  storyType === type.type
                    ? 'bg-velvet-pink-600 text-white'
                    : 'bg-black/50 text-white/70 hover:bg-black/70'
                }`}
                title={type.label}
              >
                {type.icon}
              </button>
            ))}
          </div>
        </div>

        {/* Botões de ação */}
        {(preview || file || (storyType === 'text' && text.trim())) && (
          <div className="absolute bottom-8 left-0 right-0 px-6 z-20">
            <div className="flex space-x-4">
              <button
                onClick={() => {
                  setPreview(null);
                  setFile(null);
                  setText('');
                  setLinkUrl('');
                  setLinkTitle('');
                  setLinkDescription('');
                  if (storyType === 'camera' || storyType === 'video') {
                    startCamera();
                  }
                }}
                className="flex-1 py-3 bg-gray-600 text-white rounded-lg hover:bg-gray-700 transition-colors"
              >
                Refazer
              </button>
              
              <button
                onClick={() => setCurrentStep('info')}
                className="flex-1 py-3 bg-velvet-pink-600 text-white rounded-lg hover:bg-velvet-pink-700 transition-colors"
              >
                Postar Story
              </button>
            </div>
          </div>
        )}

        {/* Hidden elements */}
        <input
          ref={fileInputRef}
          type="file"
          accept="image/*,video/*"
          onChange={handleFileSelect}
          className="hidden"
        />
        
        <canvas ref={canvasRef} className="hidden" />
      </div>

    </div>
  );

};

export default CreateStoryModal;
