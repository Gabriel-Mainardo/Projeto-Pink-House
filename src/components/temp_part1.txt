  const generateTextImage = (): Promise<File> => {
    return new Promise((resolve) => {
      const canvas = canvasRef.current!;
      const ctx = canvas.getContext('2d')!;
      
      // Configurar canvas (9:16 aspect ratio para stories)
      canvas.width = 480;
      canvas.height = 854;
      
      // Fundo

      ctx.fillRect(0, 0, canvas.width, canvas.height);
      
      // Configurar texto
      ctx.fillStyle = textColor;
      ctx.font = 'bold 32px Arial, sans-serif';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      
      // Quebrar texto em linhas
      const words = text.split(' ');
      const lines = [];
      let currentLine = '';
      
      for (const word of words) {
        const testLine = currentLine + (currentLine ? ' ' : '') + word;
        const metrics = ctx.measureText(testLine);
        
        if (metrics.width > canvas.width - 60 && currentLine !== '') {
          lines.push(currentLine);
          currentLine = word;
        } else {
          currentLine = testLine;
        }
      }
      lines.push(currentLine);
      
      // Desenhar linhas centralizadas

      const totalHeight = lines.length * lineHeight;
      const startY = (canvas.height - totalHeight) / 2 + lineHeight / 2;
      
      lines.forEach((line, index) => {
        ctx.fillText(line, canvas.width / 2, startY + (index * lineHeight));
      });
      
      // Converter para arquivo

        if (blob) {
          const textFile = new File([blob], `text_story_${Date.now()}.png`, {
            type: 'image/png'

          resolve(textFile);

  // Função para gerar thumbnail de vídeo
  const generateVideoThumbnail = (videoFile: File): Promise<File> => {
    return new Promise((resolve, reject) => {
      const video = document.createElement('video');
      const canvas = document.createElement('canvas');
      const ctx = canvas.getContext('2d');
      
      if (!ctx) {
        reject(new Error('Não foi possível criar contexto do canvas'));
        return;
      }

      video.muted = true;
      video.crossOrigin = 'anonymous';
      
      video.onloadedmetadata = () => {
        // Definir tamanho do canvas baseado no vídeo (máximo 500x500 para thumbnail)
        const maxSize = 500;
        let { videoWidth, videoHeight } = video;
        
        if (videoWidth > videoHeight) {
          if (videoWidth > maxSize) {
            videoHeight = (videoHeight * maxSize) / videoWidth;
            videoWidth = maxSize;
          }
        } else {
          if (videoHeight > maxSize) {
            videoWidth = (videoWidth * maxSize) / videoHeight;
            videoHeight = maxSize;
          }
        }
        
        canvas.width = videoWidth;
        canvas.height = videoHeight;
        
        // Ir para o primeiro segundo do vídeo para capturar uma boa imagem
        video.currentTime = Math.min(1, video.duration / 2);
      };
      
      video.onseeked = () => {
        try {
          // Desenhar o frame do vídeo no canvas
          ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
          
          // Converter canvas para blob com qualidade reduzida para thumbnail
          canvas.toBlob((blob) => {
            if (blob) {
              const thumbnailFile = new File([blob], `thumbnail_${Date.now()}.jpg`, { 
                type: 'image/jpeg' 
              });
              console.log('✅ Thumbnail gerada:', thumbnailFile.name, thumbnailFile.size);
              resolve(thumbnailFile);
            } else {
              reject(new Error('Falha ao gerar thumbnail'));
            }
          }, 'image/jpeg', 0.7); // Reduzir qualidade para 70%
        } catch (error) {
          reject(error);
        }
      };
      
      video.onerror = () => {
        reject(new Error('Erro ao carregar vídeo para thumbnail'));
      };
      
      // Criar URL do vídeo e carregar
      const videoUrl = URL.createObjectURL(videoFile);
      video.src = videoUrl;
      video.load();
    });
